// devai-continuous.js // Single-file continuous-learning prototype (local). // // Features: // - Frontend served at / (simple page) to send prompt, save, query memory. // - Backend endpoints protected by x-api-key: /api/chat, /api/memory/add, /api/memory/query, /api/admin/trigger-train // - Local vector store persisted to memory.json (embeddings via OpenAI). // - Background worker that aggregates new items into fine_tune_data.jsonl periodically. // - Optional automatic fine-tune (requires OpenAI API support + explicit env flags). // // Setup: // 1) npm init -y // 2) npm install express body-parser cors dotenv openai uuid // 3) create .env with: // OPENAI_API_KEY=sk-... // SERVER_API_KEY=any-secret-you-chose // PORT=8080 // EMBED_MODEL=text-embedding-3-small // CHAT_MODEL=gpt-4o-mini // TRAIN_INTERVAL_MS=86400000 # default 24h // MIN_TRAIN_EXAMPLES=20 // ALLOW_AUTO_FINE_TUNE=false // CONFIRM_FINE_TUNE=no // // 4) node devai-continuous.js // // Security note: Do NOT expose to the public without proper hardening (HTTPS, auth, rate limiting). require('dotenv').config(); const fs = require('fs'); const path = require('path'); const express = require('express'); const bodyParser = require('body-parser'); const cors = require('cors'); const { v4: uuidv4 } = require('uuid'); const { Configuration, OpenAIApi } = require('openai'); const PORT = process.env.PORT || 8080; const OPENAI_KEY = process.env.OPENAI_API_KEY || ''; const SERVER_API_KEY = process.env.SERVER_API_KEY || ''; const EMBED_MODEL = process.env.EMBED_MODEL || 'text-embedding-3-small'; const CHAT_MODEL = process.env.CHAT_MODEL || 'gpt-4o-mini'; const MEMORY_FILE = path.join(__dirname, 'memory.json'); const TRAINING_FILE = path.join(__dirname, 'fine_tune_data.jsonl'); const TRAIN_INTERVAL_MS = parseInt(process.env.TRAIN_INTERVAL_MS || String(1000 * 60 * 60 * 24)); // default 24h const MIN_TRAIN_EXAMPLES = parseInt(process.env.MIN_TRAIN_EXAMPLES || '20', 10); const ALLOW_AUTO_FINE_TUNE = (process.env.ALLOW_AUTO_FINE_TUNE || 'false').toLowerCase() === 'true'; const CONFIRM_FINE_TUNE = (process.env.CONFIRM_FINE_TUNE || 'no').toLowerCase() === 'yes'; if (!OPENAI_KEY) console.warn('⚠️ OPENAI_API_KEY not set. Embeddings/chat will fail until you set it.'); if (!SERVER_API_KEY) console.warn('⚠️ SERVER_API_KEY not set. API endpoints will be unprotected until configured.'); const openai = new OpenAIApi(new Configuration({ apiKey: OPENAI_KEY })); // Utilities: local memory (persisted) let memory = []; // each item: { id, text, embedding, metadata: { createdAt, source } } try { if (fs.existsSync(MEMORY_FILE)) { memory = JSON.parse(fs.readFileSync(MEMORY_FILE, 'utf8') || '[]'); console.log(`Loaded ${memory.length} memory items from ${MEMORY_FILE}`); } } catch (e) { console.warn('Could not load memory file:', e.message); } let lastTrainingTimestamp = 0; if (fs.existsSync(TRAINING_FILE)) { lastTrainingTimestamp = fs.statSync(TRAINING_FILE).mtimeMs; } function persistMemory() { try { fs.writeFileSync(MEMORY_FILE, JSON.stringify(memory, null, 2), 'utf8'); } catch (e) { console.error('Persist memory failed:', e.message); } } function dot(a, b) { let s = 0; for (let i = 0; i < Math.min(a.length, b.length); i++) s += a[i] * b[i]; return s; } function norm(a) { return Math.sqrt(dot(a, a)) || 1; } function cosine(a, b) { return dot(a, b) / (norm(a) * norm(b)); } // OpenAI helpers async function getEmbeddingOpenAI(text) { if (!OPENAI_KEY) throw new Error('OPENAI_API_KEY not configured'); const resp = await openai.createEmbedding({ model: EMBED_MODEL, input: text }); return resp.data.data[0].embedding; } async function chatOpenAI(messages, model = CHAT_MODEL) { if (!OPENAI_KEY) throw new Error('OPENAI_API_KEY not configured'); const resp = await openai.createChatCompletion({ model, messages, temperature: 0.2, max_tokens: 800 }); return resp.data.choices[0].message.content; } // Memory ops async function addMemoryItem(text, metadata = {}) { const emb = await getEmbeddingOpenAI(text); const item = { id: uuidv4(), text, embedding: emb, metadata: { ...metadata, createdAt: Date.now() } }; memory.push(item); persistMemory(); return item.id; } async function queryMemory(queryText, topK = 5) { if (!memory.length) return []; const qEmb = await getEmbeddingOpenAI(queryText); const scored = memory.map(m => ({ id: m.id, text: m.text, metadata: m.metadata, score: cosine(qEmb, m.embedding || []) })); scored.sort((a, b) => b.score - a.score); return scored.slice(0, topK); } // Background trainer: prepare fine_tune_data.jsonl from new memory items async function buildTrainingDataAndMaybeFineTune() { console.log(`[worker] Building training data — memory size: ${memory.length}`); // Collect only items added since lastTrainingTimestamp (simple heuristic) const newItems = memory.filter(m => (m.metadata && m.metadata.createdAt) && m.metadata.createdAt > lastTrainingTimestamp); // also include older items if small const candidates = newItems.length ? newItems : memory.slice(-1000); // Extract candidate training pairs: items whose text includes both USER: and ASSISTANT: const pairs = []; for (const it of candidates) { const t = it.text || ''; if (t.startsWith('USER:') && t.includes('ASSISTANT:')) { const [u, a] = t.split('ASSISTANT:'); const prompt = u.replace(/^USER:\s*/i, '').trim(); const completion = a.trim(); if (prompt && completion) pairs.push({ prompt, completion }); } } // Deduplicate by hash of prompt+completion const seen = new Set(); const uniq = []; for (const p of pairs) { const key = (p.prompt + '||' + p.completion).slice(0, 3000); if (!seen.has(key)) { seen.add(key); uniq.push(p); } } if (uniq.length < MIN_TRAIN_EXAMPLES) { console.log(`[worker] Found ${uniq.length} training pairs (< MIN_TRAIN_EXAMPLES=${MIN_TRAIN_EXAMPLES}). Writing file but skipping auto-fine-tune.`); } else { console.log(`[worker] Found ${uniq.length} training pairs (>= MIN_TRAIN_EXAMPLES=${MIN_TRAIN_EXAMPLES}).`); } // Write JSONL const w = fs.createWriteStream(TRAINING_FILE); for (const p of uniq) { const obj = { prompt: p.prompt + '\n\n###\n\n', completion: ' ' + p.completion + ' ' }; w.write(JSON.stringify(obj) + '\n'); } w.end(); lastTrainingTimestamp = Date.now(); console.log(`[worker] Wrote ${uniq.length} training pairs to ${TRAINING_FILE}`); // Optionally attempt auto fine-tune (protected by two env flags) if (ALLOW_AUTO_FINE_TUNE && CONFIRM_FINE_TUNE && uniq.length >= MIN_TRAIN_EXAMPLES) { console.log('[worker] ALLOW_AUTO_FINE_TUNE and CONFIRM_FINE_TUNE set: attempting to upload & create fine-tune job (may fail depending on provider API).'); try { // OLD OpenAI legacy flow: createFile + createFineTune (may require different API for newer SDKs) // We'll attempt to use the "files" and "fineTunes" endpoints via openai.request if available. // Use cautious approach: upload file as multipart/form-data (SDK helper may differ). Here we'll fallback to logging instructions. console.log('[worker] Attempting automatic fine-tune is not implemented in this prototype to avoid destructive operations. Please run fine-tune manually using the generated file:', TRAINING_FILE); // If you want this automated, I can add implementation for your provider after you confirm provider/version. } catch (e) { console.warn('[worker] Auto fine-tune failed:', e.message); } } else { if (ALLOW_AUTO_FINE_TUNE) { console.log('[worker] Auto fine-tune flags require CONFIRM_FINE_TUNE=yes to proceed.'); } } } // schedule worker setInterval(() => { buildTrainingDataAndMaybeFineTune().catch(e => console.warn('[worker] error', e.message)); }, TRAIN_INTERVAL_MS); // ---------- Express server and simple frontend ---------- const app = express(); app.use(cors()); app.use(bodyParser.json({ limit: '2mb' })); function requireServerKey(req, res, next) { if (!SERVER_API_KEY) return res.status(500).json({ error: 'SERVER_API_KEY not configured on server' }); const key = req.headers['x-api-key'] || req.query.api_key; if (key !== SERVER_API_KEY) return res.status(401).json({ error: 'Unauthorized - provide x-api-key' }); next(); } app.get('/', (req, res) => { res.setHeader('Content-Type', 'text/html; charset=utf-8'); res.send(INDEX_HTML); }); // Add memory (client -> server) app.post('/api/memory/add', requireServerKey, async (req, res) => { try { const { text, metadata } = req.body; if (!text) return res.status(400).json({ error: 'text required' }); const id = await addMemoryItem(text, metadata || {}); return res.json({ ok: true, id }); } catch (e) { console.error('/api/memory/add', e.message); return res.status(500).json({ error: e.message }); } }); // Query memory app.post('/api/memory/query', requireServerKey, async (req, res) => { try { const { query, topK = 5 } = req.body; if (!query) return res.status(400).json({ error: 'query required' }); const matches = await queryMemory(query, topK); return res.json({ matches }); } catch (e) { console.error('/api/memory/query', e.message); return res.status(500).json({ error: e.message }); } }); // Chat (RAG + chat + auto-save) app.post('/api/chat', requireServerKey, async (req, res) => { try { const { messages = [], topK = 5, auto_save = true, model } = req.body; if (!Array.isArray(messages) || messages.length === 0) return res.status(400).json({ error: 'messages required' }); const userMessage = messages.slice().reverse().find(m => m.role === 'user')?.content || messages[messages.length - 1].content; let retrievedText = ''; try { const matches = await queryMemory(userMessage, topK); if (matches && matches.length) { retrievedText = matches.map(m => `Context (score:${(m.score||0).toFixed(3)}): ${m.text}`).join('\n---\n'); } } catch (e) { console.warn('RAG retrieval failed:', e.message); } const systemMessages = [{ role: 'system', content: 'أنت مساعد تقني. استخدم أي سياق مرفق لتحسين الإجابة.' }]; if (retrievedText) systemMessages.push({ role: 'system', content: 'سياقات مسترجعة:\n' + retrievedText }); const finalMessages = [...systemMessages, ...messages]; const chatModel = model || CHAT_MODEL; const assistantText = await chatOpenAI(finalMessages, chatModel); if (auto_save) { try { const combined = `USER: ${userMessage}\nASSISTANT: ${assistantText}`; await addMemoryItem(combined, { source: 'auto' }); } catch (e) { console.warn('Auto-save failed:', e.message); } } return res.json({ response: assistantText }); } catch (e) { console.error('/api/chat', e.message); return res.status(500).json({ error: e.message }); } }); // Admin: trigger training manually (protected) app.post('/api/admin/trigger-train', requireServerKey, async (req, res) => { try { await buildTrainingDataAndMaybeFineTune(); return res.json({ ok: true, msg: 'Training task executed (file generated or auto-fine-tune attempted based on flags).' }); } catch (e) { return res.status(500).json({ error: e.message }); } }); app.get('/health', (req, res) => res.json({ ok: true })); app.listen(PORT, () => { console.log(`DevAI continuous server running at http://localhost:${PORT}/`); console.log('Memory file:', MEMORY_FILE, 'Training file:', TRAINING_FILE); console.log(`TRAIN_INTERVAL_MS=${TRAIN_INTERVAL_MS} MIN_TRAIN_EXAMPLES=${MIN_TRAIN_EXAMPLES}`); }); // ----------------- Embedded simple frontend ----------------- const INDEX_HTML = ` DevAI Continuous — محلي
